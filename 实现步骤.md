## 第二步：实现 Schema 检索模块

**核心逻辑**：表级向量化（表名 + 描述 + 标签 + 字段注释聚合），命中后精准加载完整字段结构。

```python
# agent/schema_retriever.py
import chromadb

class SchemaRetriever:
    def __init__(self):
        self.client = chromadb.PersistentClient(path="vectors/table_index.db")
        self.collection = self.client.get_or_create_collection("tables")

    def build_index(self, tables: list[dict]):
        """全量构建表级向量索引"""
        for t in tables:
            doc = self._build_doc(t)
            self.collection.upsert(
                ids=[t["name"]],
                documents=[doc],
                metadatas={"source": t["source"]}
            )

    def _build_doc(self, t: dict) -> str:
        fields_text = ", ".join(t.get("comment", []))
        return f"表名:{t['name']} 描述:{t['desc']} 标签:{t['label']} 字段:{fields_text}"

    def retrieve(self, question: str, top_n: int = 5) -> list[dict]:
        results = self.collection.query(query_texts=[question], n_results=top_n)
        # 返回完整 schema（从 yaml 精准加载）
        return [load_full_schema(name) for name in results["ids"][0]]
```

### 向量更新策略

| 时机 | 方式 | 触发 |
|---|---|---|
| 用户建表/删表/改表时 | 增量 upsert | 表变更操作的调用处 |
| 定时兜底 | 全量同步 | 每天凌晨 cron，补漏+降噪 |

## 第三步：实现 NL2SQL 生成

**Prompt 结构**：方言说明（固定）+ 检索到的表结构（动态）+ 用户问题。

```python
# agent/nl2sql.py

DIALECT_PROMPT = """
数据库类型: Apache Doris
方言注意事项:
1. 时间函数: date_trunc('day', dt)
2. 股票代码格式: 000001.SZ
3. 分区查询建议带分区键过滤
4. LIMIT 语法: LIMIT 10
"""

SYSTEM_PROMPT = f"""你是SQL专家，只生成只读SELECT语句。
{DIALECT_PROMPT}
规则:
- 仅允许 SELECT 和 CTE
- 必须包含 LIMIT (默认1000)
- 禁止 DDL/DML/多语句
- 只输出SQL，不要解释
"""

async def generate_sql(question: str, schemas: list[dict]) -> str:
    schema_text = format_schemas_for_prompt(schemas)
    messages = [
        {"role": "system", "content": SYSTEM_PROMPT},
        {"role": "user", "content": f"表结构:\n{schema_text}\n\n问题: {question}"}
    ]
    raw = await call_llm(messages)
    sql = extract_sql_from_response(raw)
    return sql
```

### 方言说明不做向量库的理由

- 量级小（几 KB），全局适用，每次都要带上
- 变更频率极低（数据库版本升级才变）
- LLM 已内置通用 SQL 知识，只需补充差异点

## 第四步：实现 SQL 安全校验

使用 `sqlglot` 做 AST 解析，确保仅 SELECT、无危险操作、必须 LIMIT。

```python
# agent/sql_validator.py
import sqlglot
from sqlglot import exp

FORBIDDEN_NODES = (exp.Insert, exp.Update, exp.Delete,
                   exp.Drop, exp.Alter, exp.Create)

def validate_readonly(sql: str) -> str:
    # 1. 禁止多语句
    statements = sqlglot.parse(sql)
    if len(statements) != 1:
        raise ValueError("禁止多语句")

    ast = statements[0]

    # 2. 必须是 SELECT 或 WITH...SELECT
    if not isinstance(ast, (exp.Select, exp.With)):
        raise ValueError("仅允许SELECT")

    # 3. 禁止危险节点
    for node in ast.walk():
        if isinstance(node, FORBIDDEN_NODES):
            raise ValueError(f"禁止的操作: {type(node).__name__}")

    # 4. 必须有 LIMIT，没有则自动补
    if not ast.args.get("limit"):
        sql = f"{sql.rstrip(';')} LIMIT 1000"

    # 5. 引用的表必须在白名单中
    referenced = [t.name for t in ast.find_all(exp.Table)]
    # 校验 referenced ⊆ authorized_tables

    return sql
```

### 校验规则清单（对应产品文档 7.1 节）

- 仅允许 `SELECT` 和只读 CTE
- 禁止多语句
- 禁止注释绕过
- 禁止 `INSERT`、`UPDATE`、`DELETE`、`MERGE`、DDL、授权命令
- 禁止调用未授权函数、外部文件和网络函数
- 必须包含最大行数限制
- 必须限制执行时间、扫描行数和内存
- 必须通过 SQL AST 判断，不得仅使用字符串匹配
- 引用对象必须全部在权限白名单中

## 第五步：实现 SQL 执行 + Python 函数执行

```python
# agent/executor.py
import json
import pandas as pd
import numpy as np

class Executor:
    def __init__(self, db_sdk):
        self.db_sdk = db_sdk  # 已有的 Python SDK

    async def execute_sql(self, sql: str) -> dict:
        sql = validate_readonly(sql)
        df = self.db_sdk.query(sql)  # SDK 返回 DataFrame
        return {
            "columns": df.columns.tolist(),
            "rows": df.values.tolist(),
            "rowCount": len(df),
        }

    async def execute_python(self, code: str, data: str = None) -> dict:
        """在受限环境中执行 Python 数据处理"""
        local_vars = {}
        if data:
            df = pd.DataFrame(json.loads(data))
            local_vars["df"] = df

        exec_globals = {
            "pd": pd, "np": np,
            "__builtins__": safe_builtins(),
        }
        exec(code, exec_globals, local_vars)

        result = local_vars.get("result")
        if isinstance(result, pd.DataFrame):
            return {
                "columns": result.columns.tolist(),
                "rows": result.values.tolist(),
                "rowCount": len(result),
            }
        return {"value": str(result)}
```

### Python 执行安全策略

- 白名单导入：仅允许 `pandas`、`numpy`、`json`、`math` 等
- 禁止 `os`、`subprocess`、`open`、`eval`、`exec` 等危险内置
- 执行超时限制（如 30 秒）
- 内存限制

## 第六步：实现 ECharts 配置生成

```python
# agent/chart_generator.py
import pandas as pd

async def generate_chart(chart_type: str, data: dict, title: str = "") -> dict:
    df = pd.DataFrame(data["rows"], columns=data["columns"])

    if chart_type == "line":
        return _build_line_chart(df, title)
    elif chart_type == "bar":
        return _build_bar_chart(df, title)
    elif chart_type == "pie":
        return _build_pie_chart(df, title)
    elif chart_type == "scatter":
        return _build_scatter_chart(df, title)
    elif chart_type == "heatmap":
        return _build_heatmap_chart(df, title)
    else:
        raise ValueError(f"不支持的图表类型: {chart_type}")

def _build_line_chart(df, title):
    x_field = df.columns[0]
    y_fields = [c for c in df.columns if c != x_field]
    return {
        "title": {"text": title},
        "tooltip": {"trigger": "axis"},
        "xAxis": {"type": "category", "data": df[x_field].tolist()},
        "yAxis": {"type": "value"},
        "series": [
            {"name": f, "type": "line", "data": df[f].tolist()}
            for f in y_fields
        ]
    }

def _build_bar_chart(df, title):
    x_field = df.columns[0]
    y_fields = [c for c in df.columns if c != x_field]
    return {
        "title": {"text": title},
        "tooltip": {"trigger": "axis"},
        "xAxis": {"type": "category", "data": df[x_field].tolist()},
        "yAxis": {"type": "value"},
        "series": [
            {"name": f, "type": "bar", "data": df[f].tolist()}
            for f in y_fields
        ]
    }

def _build_pie_chart(df, title):
    name_field = df.columns[0]
    value_field = df.columns[1]
    return {
        "title": {"text": title},
        "tooltip": {"trigger": "item"},
        "series": [{
            "type": "pie",
            "data": [
                {"name": str(row[name_field]), "value": row[value_field]}
                for _, row in df.iterrows()
            ]
        }]
    }

def _build_scatter_chart(df, title):
    x_field = df.columns[0]
    y_field = df.columns[1]
    return {
        "title": {"text": title},
        "tooltip": {"trigger": "item"},
        "xAxis": {"type": "value", "name": x_field},
        "yAxis": {"type": "value", "name": y_field},
        "series": [{
            "type": "scatter",
            "data": df[[x_field, y_field]].values.tolist()
        }]
    }

def _build_heatmap_chart(df, title):
    x_field = df.columns[0]
    y_field = df.columns[1]
    value_field = df.columns[2]
    x_categories = df[x_field].unique().tolist()
    y_categories = df[y_field].unique().tolist()
    data = []
    for _, row in df.iterrows():
        x_idx = x_categories.index(row[x_field])
        y_idx = y_categories.index(row[y_field])
        data.append([x_idx, y_idx, row[value_field]])
    return {
        "title": {"text": title},
        "tooltip": {"trigger": "item"},
        "xAxis": {"type": "category", "data": x_categories},
        "yAxis": {"type": "category", "data": y_categories},
        "visualMap": {
            "min": df[value_field].min(),
            "max": df[value_field].max(),
            "calculable": True,
            "orient": "horizontal",
            "left": "center",
            "bottom": "15"
        },
        "series": [{"type": "heatmap", "data": data}]
    }
```

### 图表自动推荐规则

| 数据形态 | 默认图表 |
|---|---|
| 时间 + 数值 | 折线图 (line) |
| 分类 + 数值 | 柱状图 (bar) |
| 占比数据 | 饼图 (pie) |
| 两个连续变量 | 散点图 (scatter) |
| 股票 + 涨跌 | 热力图 (heatmap) |
| 不适合可视化 | 不自动生成 |

## 第七步：组装 MCP Server

```python
# mcp_server.py
from mcp.server.fastmcp import FastMCP
from agent.schema_retriever import SchemaRetriever
from agent.nl2sql import generate_sql
from agent.sql_validator import validate_readonly
from agent.executor import Executor
from agent.chart_generator import generate_chart
import json

mcp = FastMCP("sql-agent")

retriever = SchemaRetriever()
executor = Executor(db_sdk=your_sdk_instance)

@mcp.tool()
async def search_tables(question: str) -> str:
    """根据自然语言问题检索相关数据库表结构"""
    tables = retriever.retrieve(question, top_n=5)
    return json.dumps(tables, ensure_ascii=False)

@mcp.tool()
async def generate_and_execute_sql(question: str) -> str:
    """从自然语言生成SQL并执行，返回查询结果"""
    schemas = retriever.retrieve(question, top_n=5)
    sql = await generate_sql(question, schemas)
    sql = validate_readonly(sql)
    result = await executor.execute_sql(sql)
    return json.dumps({"sql": sql, "result": result}, ensure_ascii=False)

@mcp.tool()
async def generate_echarts_config(chart_type: str, data: str, title: str = "") -> str:
    """将数据转换为ECharts配置JSON"""
    parsed = json.loads(data)
    config = await generate_chart(chart_type, parsed, title)
    return json.dumps(config, ensure_ascii=False)

if __name__ == "__main__":
    mcp.run(transport="stdio")
```

在 `~/.config/opencode/skills/sql-agent/SKILL.md` 中编写 Skill，引导 Agent 按流程编排 MCP 工具：

```markdown
# SQL Agent Skill

## 触发条件
用户需要查询数据库、生成图表、分析数据时触发

## 执行流程
1. 调用 `search_tables` 检索相关表结构
2. 调用 `generate_and_execute_sql` 生成SQL并执行
3. 若需数据处理，调用 `execute_python` 用 pandas 加工
4. 若需图表，调用 `generate_echarts_config` 生成 ECharts 配置
5. 将结果以表格 + 图表 + 文字结论返回用户

## 规则
- 只允许 SELECT 查询
- 图表类型自动判断：时间+数值→折线，分类+数值→柱状，占比→饼图
- 大数据量提示用户缩小范围
```
